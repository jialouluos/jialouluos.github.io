( line = fgets( buffer ) ) ) { return rgbe_error( rgbe_read_error, 'no header found' ); } /* if you want to require the magic token then uncomment the next line */ if ( ! byteArray.pos = 0; const rgbe_header_info = RGBE_ReadHeader( byteArray ); if ( RGBE_RETURN_FAILURE !== rgbe_header_info ) { const w = rgbe_header_info.width, h = rgbe_header_info.height, image_rgba_data = RGBE_ReadPixels_RLE( byteArray.subarray( byteArray.pos ), w, h );
